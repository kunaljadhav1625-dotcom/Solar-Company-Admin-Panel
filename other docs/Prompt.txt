        // Password visibility toggle
        function togglePassword() {
            const passwordInput = document.getElementById('password');
            const toggleBtn = document.querySelector('.password-toggle');
            
            if (passwordInput.type === 'password') {
                passwordInput.type = 'text';

        // Form submission handling
        document.getElementById('loginForm').addEventListener('submit', function(e) {
            e.preventDefault();
            
            const username = document.getElementById('username').value.trim();
            const password = document.getElementById('password').value.trim();
            const remember = document.getElementById('remember').checked;
            const loginButton = document.getElementById('loginButton');
            const loadingIndicator = document.getElementById('loadingIndicator');
            const errorAlert = document.getElementById('errorAlert');
            const successAlert = document.getElementById('successAlert');
            
            // Hide previous alerts
            errorAlert.style.display = 'none';
            successAlert.style.display = 'none';
            errorAlert.classList.remove('show');
            successAlert.classList.remove('show');
            
            // Basic validation
            if (!username || !password) {
                showError('Please fill in all fields.');
                return;
            }
            
            if (password.length < 6) {
                showError('Password must be at least 6 characters long.');
                return;
            }
            
            // Show loading state
            loginButton.style.display = 'none';
            loadingIndicator.style.display = 'block';
            loginButton.disabled = true;
            
            // Prepare form data
            const formData = {
                username: username,
                password: password,
                remember: remember
            };
            
            // Send login request
            fetch('login.php', {
                method: 'POST',
                headers: {
                    'Content-Type': 'application/json'
                },
                body: JSON.stringify(formData)
            })
            .then(response => response.json())
            .then(data => {
                if (data.success) {
                    showSuccess(data.message || 'Login successful! Redirecting to dashboard...');
                    
                    // Store login info in localStorage for frontend
                    localStorage.setItem('adminLoggedIn', 'true');
                    localStorage.setItem('adminUsername', data.data?.user?.email || username);
                    
                    // Redirect after delay
                    setTimeout(() => {
                        window.location.href = data.data?.redirect || 'admin-dashboard.php';
                    }, 1500);
                } else {
                    showError(data.message || 'Login failed. Please try again.');
                    resetLoginForm();
                }
            })
            .catch(error => {
                console.error('Login error:', error);
                showError('Network error. Please check your connection and try again.');
                resetLoginForm();
            });
        });
        
        function showError(message) {
            const errorAlert = document.getElementById('errorAlert');
            errorAlert.textContent = message;
            errorAlert.style.display = 'block';
            errorAlert.classList.add('show');
            
            // Auto-hide after 5 seconds
            setTimeout(() => {
                errorAlert.style.display = 'none';
                errorAlert.classList.remove('show');
            }, 5000);
        }
        
        function showSuccess(message) {
            const successAlert = document.getElementById('successAlert');
            successAlert.textContent = message;
            successAlert.style.display = 'block';
            successAlert.classList.add('show');
        }
        
        function resetLoginForm() {
            const loginButton = document.getElementById('loginButton');
